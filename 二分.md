int search(int r, int l){

    while (l <= r){

        int mid = (r-l)/2 + l;

        int ans = -1;

        if (check(mid)) {

            ans = mid;

            l = mid + 1; //找最大值，继续往右边找

            //r = mid - 1 //找最小值，继续往左边找

        }

        else{

            r = mid - 1; //根据单调性书写 l = mid + 1;

        }

    }

    return ans;

}
