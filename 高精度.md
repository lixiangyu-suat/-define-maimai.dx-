struct BigInt {
    vector<int> d;
    
    // 构造：支持 int 直接初始化
    BigInt(long long x = 0) {
        if (!x) d.push_back(0);
        while (x) d.push_back(x % 10), x /= 10;
    }

    // 比较：先长度后位值
    bool operator < (const BigInt& b) const {
        if (d.size() != b.d.size()) return d.size() < b.d.size();
        for (int i = d.size() - 1; i >= 0; i--)
            if (d[i] != b.d[i]) return d[i] < b.d[i];
        return false;
    }

    // 高精 * 低精
    BigInt operator * (const int& x) const {
        BigInt res; res.d.clear();
        long long t = 0;
        for (int i = 0; i < d.size() || t; i++) {
            if (i < d.size()) t += (long long)d[i] * x;
            res.d.push_back(t % 10);
            t /= 10;
        }
        return res;
    }

    // 高精 * 高精 (模拟竖式乘法)
    BigInt operator * (const BigInt& b) const {
        BigInt res;
        res.d.assign(d.size() + b.d.size(), 0);
        for (int i = 0; i < d.size(); i++) {
            for (int j = 0; j < b.d.size(); j++) {
                long long t = res.d[i + j] + (long long)d[i] * b.d[j];
                res.d[i + j] = t % 10;
                res.d[i + j + 1] += t / 10; // 处理进位
            }
        }
        while (res.d.size() > 1 && res.d.back() == 0) res.d.pop_back();
        return res;
    }

    // 高精 / 低精
    BigInt operator / (const int& x) const {
        BigInt res; res.d.clear();
        long long r = 0;
        for (int i = d.size() - 1; i >= 0; i--) {
            r = r * 10 + d[i];
            res.d.push_back(r / x);
            r %= x;
        }
        reverse(res.d.begin(), res.d.end());
        while (res.d.size() > 1 && res.d.back() == 0) res.d.pop_back();
        return res;
    }

    // 输出结果
    void print() {
        for (int i = d.size() - 1; i >= 0; i--) cout << d[i];
        cout << endl;
    }
};
