#include<bits/stdc++.h>
#define int long long
using namespace std;
const int N = 1e4 + 5;
int dp[N], n, m, C;
signed main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin>>n>>m>>C;
    for(int i=1;i<=n;i++){
        int v, w, d;
        cin>>v>>w>>d;
        if(d*v > C) d = C/v;
        int v_ = v, w_ = w;
        int num = 1, sum = 0;
        while(sum*2 < d){
            sum += num;
            for(int i=C;i>=v;i--) dp[i] = max(dp[i], dp[i-v] + w);
            v *= 2; w *= 2;
            num <<= 1;
        }
        v = v_*(d-sum); w = w_*(d-sum);
        if(v && w) for(int i=C;i>=v;i--) dp[i] = max(dp[i], dp[i-v] + w);
    }
    for(int i=1;i<=m;i++){
        int a2, a1, a0;
        cin>>a2>>a1>>a0;
        for(int j=C;j>=0;j--){
            for(int k=0;k<=j;k++){
                dp[j] = max(dp[j], dp[j-k] + a2*k*k + a1*k + a0);
            }
        }
    }
    cout<<dp[C];
}
